<script>
/* ===== Pomodoro ===== */
let timerInterval;
let time; // tiden i sekunder

function startTimer(minutes){
    clearInterval(timerInterval);
    time = minutes * 60; // sätter time i sekunder

    timerInterval = setInterval(()=>{
        let mins = Math.floor(time/60);
        let secs = time % 60;
        document.getElementById("timer").innerText =
            mins.toString().padStart(2,'0') + ":" + secs.toString().padStart(2,'0');
        time--;
        if(time < 0){
            clearInterval(timerInterval);
            timerInterval = null;
            document.getElementById("alarm-sound").play();
            alert("Tiden är slut!");
        }
    }, 1000);
}

function startStudy(){
    let studyMinutes = parseInt(document.getElementById("study").value) || 25;
    startTimer(studyMinutes);
}

function startBreak(){
    let breakMinutes = parseInt(document.getElementById("break").value) || 5;
    startTimer(breakMinutes);
}

// ✅ Ny reset-funktion
function resetTimer() {
    clearInterval(timerInterval);  // Stoppar aktuell timer
    timerInterval = null;

    // Läs in plugg-tid
    let studyMinutes = parseInt(document.getElementById("study").value) || 25;
    time = studyMinutes * 60; // Sätt time tillbaka i sekunder

    // Uppdatera displayen
    document.getElementById("timer").innerText =
        studyMinutes.toString().padStart(2,'0') + ":00";
}

/* ===== Miniräknare ===== */
function addCalc(value) {
    document.getElementById("calc-display").value += value;
}

function calculate() {
    try {
        document.getElementById("calc-display").value =
            eval(document.getElementById("calc-display").value);
    } catch {
        alert("Felaktigt uttryck");
    }
}

function clearCalc() {
    document.getElementById("calc-display").value = "";
}

/* ===== Kom ihåg-lista ===== */
function loadRemember(){
    const items = JSON.parse(localStorage.getItem('rememberItems') || '[]');
    const list = document.getElementById('remember-list');
    list.innerHTML = '';
    items.forEach((item,index)=>{
        const li = document.createElement('li');
        const checkbox = document.createElement('input');
        checkbox.type = 'checkbox';
        checkbox.checked = item.checked;
        checkbox.onchange = ()=>{
            items[index].checked = checkbox.checked;
            localStorage.setItem('rememberItems', JSON.stringify(items));
        };
        li.appendChild(checkbox);

        const span = document.createElement('span');
        span.innerText = item.text;
        li.appendChild(span);

        const removeBtn = document.createElement('button');
        removeBtn.innerText = 'Ta bort';
        removeBtn.onclick = ()=>{
            items.splice(index,1);
            localStorage.setItem('rememberItems', JSON.stringify(items));
            loadRemember();
        };
        li.appendChild(removeBtn);

        list.appendChild(li);
    });
}

function addRemember(){
    const input = document.getElementById('remember-input');
    const text = input.value.trim();
    if(text === '') return;
    const items = JSON.parse(localStorage.getItem('rememberItems') || '[]');
    items.push({text:text,checked:false});
    localStorage.setItem('rememberItems', JSON.stringify(items));
    input.value = '';
    loadRemember();
}

// Ladda listan vid start
loadRemember();
